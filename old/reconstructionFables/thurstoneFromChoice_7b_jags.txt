# choice probabilities for perceptual reproduction task
data{
  pi = 3.1415
}
model{
  mu[1] = 0
  mu[2] ~ dunif(0, pi/2)
  for (j in 3:nStimuli){
    mu[j] ~ dunif(0, pi)
  } 
  sigmaTmp[1] ~ dunif(0, pi)
  sigmaTmp[2] ~ dunif(0, pi)
  sigmaSrt = sort(sigmaTmp)
  sigmaP = sigmaSrt[1]
  sigmaM = sigmaSrt[2]
  for (t in 1:nTrialsP){
    yP[t] ~ dnorm(mu[sP[t]], 1/sigmaP^2)
  }
  for (t in 1:nTrialsM){
    yM[t] ~ dnorm(mu[sM[t]], 1/sigmaM^2)
  }
}
