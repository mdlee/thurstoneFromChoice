# choice probabilities for memory reproduction task
data{
  pi = 3.1415
}
model{
  mu[1] = 0
  mu[2] ~ dunif(0, pi/2)
  for (j in 3:nStimuli){
    mu[j] ~ dunif(0, pi)
  } 
  sigma ~ dunif(0, pi)
  omega ~ ddirch(rep(1, nFoils+1))
  for (t in 1:nTrials){
    xT[t] ~ dnorm(mu[target[t]], 1/sigma^2)
    for (j in 1:nFoils){
      xF[t, j] ~ dnorm(foil[t, j], 1/sigma^2)
    }
    tau[t] = inprod(c(xT[t], xF[t, 1:nFoils]), omega)
    y[t] ~ dnorm(tau[t], 1/(0.01)^2)
  }
}
