# estimation in memory task
# e: swap error
data{
  pi = 3.1415
}
model{
  mu[1] = 0
  mu[2] ~ dunif(0, pi/2)
  for (j in 3:nStimuli){
    mu[j] ~ dunif(0, pi)
  } 
  sigma ~ dunif(0, pi)
  omega ~ ddirch(rep(1, setSize))
  for (t in 1:nTrials){
    z[t] ~ dcat(omega)
    y[t] ~ dnorm(mu[stim[t, z[t]]], 1/sigma^2)
  }
}
