# estimation in memory task
# e: swap error
# f: both 3 and 6, 2 drop order constraint
data{
  pi = 3.1415
}
model{
  mu[1] = 0
  mu[2] ~ dunif(0, pi/2)
  for (j in 3:nStimuli){
    mu[j] ~ dunif(0, pi)
  } 
  sigma[1] ~ dunif(0, pi)
  sigma[2] ~ dunif(0, pi)
  omega3 ~ ddirch(rep(1, 3))
  omega6 ~ ddirch(rep(1, 6))
  for (t in 1:nTrials){
    xi[t, 1] ~ dcat(omega3)
    xi[t, 2] ~ dcat(omega6)
    y[t] ~ dnorm(mu[stim[t, xi[t, setSize[t]]]], 1/sigma[setSize[t]]^2)
  }
}
