# perceptual reproduction task
data{
  pi = 3.1415
}
model{
 # identifiability priors
  mu[1] = 0
  mu[2] ~ dunif(0, pi/2)
  for (j in 3:nStimuli){
    mu[j] ~ dunif(0, pi)
  } 
  # reproduction noise
  sigma ~ dunif(0, pi)
  # experimental trials
  for (t in 1:nTrials){
    # response
    y[t] ~ dnorm(mu[s[t]], 1/sigma^2)
  }
}
