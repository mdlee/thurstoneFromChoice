# choice probabilities for "this or that" task
model{
  for (j in 1:nStimuli){
    mu[j] ~ dunif(0, 1)
  } 
  sigma ~ dunif(0, 1)
  for (t in 1:nTrials){
    xA[t] ~ dnorm(mu[a[t]], 1/sigma^2)
    xB[t] ~ dnorm(mu[b[t]], 1/sigma^2)
    xS[t] ~ dnorm(mu[s[t]], 1/sigma^2)
    y[t] ~ dinterval(abs(xA[t]-xS[t]) - abs(xB[t]-xS[t]), 0)
  }
}
